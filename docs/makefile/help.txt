#################
# Makefile HELP #
#################

- Build code:
  make build

- Build docker image:
  make docker

- Clean:
  make clean

- Test:
  make test

- Run Lithos adoption gates (static safety scan + manifest conformance):
  make lithos-verify

- All:
  make all

- Publish to testpypi:
  make publish-test

- Publish to pypi:
  make publish
